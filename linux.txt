	# Add index.php to the list if you are using PHP
	index index.php index.html index.htm index.nginx-debian.html;

	server_name _;

	location /youzhu/ {
 		if (!-e $request_filename){
          		rewrite ^/youzhu/(.*)$ /youzhu/index.php/$1 last;

     		}

		try_files $uri $uri/ =404;
